لتفعيل المشاركة مع أصدقائك، تحتاج ربط قاعدة بيانات Supabase مجانية
الخطوة 1: روح supabase.com وأنشئ حساب مجاني

الخطوة 2: أنشئ Project جديد

الخطوة 3: روح Settings → API وانسخ:
- Project URL
- anon public key

الخطوة 4: روح SQL Editor وشغّل: 
CREATE TABLE novels (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text, author text, genre text,
  description text, emoji text,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE chapters (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  novel_id uuid REFERENCES novels(id) ON DELETE CASCADE,
  title text, content text, language text,
  translation text, translated boolean DEFAULT false,
  sort_order int DEFAULT 0,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE chapter_files (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  chapter_id uuid REFERENCES chapters(id) ON DELETE CASCADE,
  name text, type text, url text, size bigint,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE novels ENABLE ROW LEVEL SECURITY;
ALTER TABLE chapters ENABLE ROW LEVEL SECURITY;
ALTER TABLE chapter_files ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public_all" ON novels FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "public_all" ON chapters FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "public_all" ON chapter_files FOR ALL USING (true) WITH CHECK (true);
